def concatenate_list_data(list,char=''):
     result= ''
     for element in list:
         if element == list[-1]:
             result += str(element)
         else:
             result += str(element)+char
     return result

print(concatenate_list_data(response.css('div.PrevAddress::text').re('\w+'),'-'))
